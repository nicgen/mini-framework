MiniFramework

A lightweight, modern JavaScript framework with Virtual DOM, state management, and routing - built from scratch for educational purposes and real-world applications.

Live Demo Run Tests

Core Features

⚡

Virtual DOM

Efficient DOM manipulation with virtual DOM diffing algorithm for optimal performance and minimal reflows.

🎯

State Management

Redux-inspired centralized state management with middleware support, dev tools integration, and persistence.

🚀

Component System

Class-based components with lifecycle hooks, local state, and reusable, composable architecture.

🛣️

Client-Side Routing

Hash and history-based routing with nested routes, guards, and automatic URL synchronization.

🎭

Event System

Custom event delegation system with automatic cleanup and support for both native and custom events.

📦

Zero Dependencies

No external dependencies - everything built from scratch using modern ES6+ JavaScript features.

Interactive Demos

import framework from './src/core/framework.js';

// Create state store for counter
const store = framework.createStore({
    count: 0
}, {
    INCREMENT: (state) => ({ ...state, count: state.count + 1 }),
    DECREMENT: (state) => ({ ...state, count: state.count - 1 })
});

// Counter component
function renderCounter() {
    const state = store.getState();

    return framework.createElement('div', {
        className: 'counter-app'
    },
        framework.createElement('h2', {}, 'Counter: ' + state.count),
        framework.createElement('button', {
            onClick: () => {
                store.dispatch({ type: 'INCREMENT' });
                update();
            }
        }, '+1')
    );
}

function update() {
    framework.render(renderCounter(), document.getElementById('root'));
}

import framework from './src/core/framework.js';

// Simple todo state management
let todos = [];
let nextId = 1;

function addTodo(text) {
    todos.push({ id: nextId++, text, completed: false });
    render();
}

function toggleTodo(id) {
    todos = todos.map(todo =>
        todo.id === id ? { ...todo, completed: !todo.completed } : todo
    );
    render();
}

function deleteTodo(id) {
    todos = todos.filter(todo => todo.id !== id);
    render();
}

function renderTodos() {
    return framework.createElement('div', {},
        framework.createElement('input', {
            placeholder: 'Add new todo...',
            onKeydown: (e) => {
                if (e.key === 'Enter' && e.target.value.trim()) {
                    addTodo(e.target.value.trim());
                    e.target.value = '';
                }
            }
        }),
        ...todos.map(todo =>
            framework.createElement('div', { key: todo.id },
                framework.createElement('span', {
                    onClick: () => toggleTodo(todo.id),
                    style: { textDecoration: todo.completed ? 'line-through' : 'none' }
                }, todo.text),
                framework.createElement('button', {
                    onClick: () => deleteTodo(todo.id)
                }, '×')
            )
        )
    );
}

import framework from './src/core/framework.js';

// Set up routing
let currentView = 'home';

framework.route('/home', () => {
    currentView = 'home';
    render();
});

framework.route('/about', () => {
    currentView = 'about';
    render();
});

framework.route('/contact', () => {
    currentView = 'contact';
    render();
});

function renderView() {
    const views = {
        home: '🏠 Welcome to the Home page!',
        about: 'ℹ️ About us: We build awesome frameworks!',
        contact: '📧 Contact: hello@miniframework.dev'
    };

    return framework.createElement('div', {},
        framework.createElement('nav', {},
            ['home', 'about', 'contact'].map(route =>
                framework.createElement('button', {
                    onClick: () => framework.navigate('/' + route),
                    className: currentView === route ? 'active' : ''
                }, route.toUpperCase())
            )
        ),
        framework.createElement('main', {}, views[currentView])
    );
}
